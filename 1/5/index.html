8. Argumentos Indefinidos (*args)

        *args = arguments, va por convención.
        Para definir un números de argumentos variables, de acuerdo al uso que le de el usuario.
        
        ejemplo
        
        def suma(*args)
            for num in args:
                print(num)
        
        suma(1)
        suma(1,2)
        suma(1,2,3,4,5,6,7,8)
        
        --------------------------------------------------------
                        
                                -> Mejor *args por convenciòn
        def suma(*cualquiercosa)			-> El nombre puede variar, siempre y cuando comienze con un *
            for num in cualquiercosa:
                print(num)
        
        suma(1)
        suma(1,2)
        suma(1,2,3,4,5,6,7,8)
        
        --------------------------------------------------------
        
        def suma(a,b)
            return a+b
        
        print(suma(5,6,5))		-> Error
        
        --------------------------------------------------------
        
        def suma(*args):
            total = 0
        
            for arg in args:
                total += arg
            return total
        
        
        print(suma(5,6,5))		-> 16
        print(suma(5,6))		-> 11
        
        
        --------------------------------------------------------
                        -> Más sencillo, y args por convenciòn.
        def suma(*args):
            return sum(args)
        
        print(suma(5,6,100,2,5,9,10))		-> 137

        def pedir_letra():
            letra_elegida = ''
            es_valida = False

                    es_valida = True
                else:
                    print("No has elegido una letra correcta")
        
            return letra_elegida
        
        
        def mostrar_nuevo_tablero(palabra_elegida):
        
            lista_oculta = []
        
            for l in palabra_elegida:
                if l in letras_correctas:
                    lista_oculta.append(l)
                else:
                    lista_oculta.append('-')
        
            print(' '.join(lista_oculta))
        
        
        def chequear_letra(letra_elegida, palabra_oculta, vidas, coincidencias):

            else:
                letras_incorrectas.append(letra_elegida)
                vidas -= 1
        
            if vidas == 0:

            elif coincidencias == letras_unicas:

            print("Felicitaciones, has encontrado la palabra!!!")
        
            return True
