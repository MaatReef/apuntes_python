6. Controlar el Movimiento

        Recordar que con pygame, cada cosa que sucede en pantalla es un evento.. O bien en teclado, etc.
        Y luego controlar el desenlace de ese evento.

        En este caso lo haremos a través del teclado.

        --------------------------------------------------------
        import pygame

        # Inicializar Pygame
        pygame.init()			


        # Crear la Pantalla
        pantalla = pygame.display.set_mode((800, 600))			


        #Título e Icono
        pygame.display.set_caption("Invasión Espacial")			
        icono = pygame.image.load("ovni.png")				
        pygame.display.set_icon(icono)


        #Jugador
        img_jugador = pygame.image.load("cohete.png")
        jugador_x = 368				
        jugador_y = 536	
        jugador_x_cambio = 0

        def jugador(x, y):						
            pantalla.blit(img_jugador, (x, y))			


        # Loop del  Juego
        se_ejecuta = True
        while se_ejecuta:						

            #RGB lo ponemos al comienzo para que luego no interfiera.
            pantalla.fill((205, 144, 228))		
                    

            for evento in pygame.event.get():			
                if evento.type == pygame.QUIT:			
                    se_ejecuta = False

                if evento.type == pygame.KEYDOWN:			-> Tecla presionada, cualquiera
                
                    if evento.key == pygame.K_LEFT:			-> Tecla izquierda presionada
                        jugador_x_cAmbio = -0.3
                    if evento.key == pygame.K_RIGHT:		-> Tecla derecha presionada.
                        jugador_x_cAmbio = 0.3
                if evento.type == pygame.KEYUP:
                    if  evento.key == pygame.K_LEFT or if evento.key == pygame.K_RIGHT:	-> Tecla Soltada
                        jugador_x_cAmbio = 0

            jugador_x += jugador_c_cambio						-> Actualizamos los datos de la posición de acuerdo a lo pulsado.
            jugador(jugador_x, jugador_y)					
            

            
            pygame.display.update()

        import pygame
        import random

        pygame.init()

        jugador_y = 500
        jugador_x_cambio = 0

        import pygame
        import random		


        # Inicializar Pygame
        pygame.init()			



        # Crear la Pantalla
        pantalla = pygame.display.set_mode((800, 600))			


        #Título e Icono
        pygame.display.set_caption("Invasión Espacial")			
        icono = pygame.image.load("ovni.png")				
        pygame.display.set_icon(icono)
        fondo = pygame.image.load('fondo.jpg')			

        # Variables del Jugador
        img_jugador = pygame.image.load("cohete.png")
        jugador_x = 368				
        jugador_y = 500
        jugador_x_cambio = 0


        # Variables del Enemigo						
        img_enemigo = pygame.image.load("enemigo.png")
        enemigo_x = random.randint(0, 736)				
        enemigo_y = random.randint(50, 200)
        enemigo_x_cambio = 1			
        enemigo_y_cambio = 50				

        # Variables de la Bala
        img_bala = pygame.image.load("bala .png")
        bala_x = 0				
        bala_y = 500			-> Altura de la nave, jugador
        bala_x_cambio = 0			
        bala_y_cambio = 1
        bala_visible = False		-> Al comienzo del juego de la bala no se verá hasta que el jugador la dispare.

        # Función jugador
        def jugador(x, y):						
            pantalla.blit(img_jugador, (x, y))			


        # Función Enemigo
        def enemigo(x, y):						
            pantalla.blit(img_enemigo, (x, y))			

        # Función disparar bala
        def disparar_bala(x, y):
            global bala_visible
            bala_visible = True
            pantalla.blit(img_bala, (x + 16, y + 10))		-> Ubicación desde el jugador


        # Loop del  Juego
        se_ejecuta = True
        while se_ejecuta:						

            # Imagen de fondo
            pantalla.blit(fondo, (0, 0))			
                    

            # Iterar eventos
            for evento in pygame.event.get():			

                # Evento cerrar
                if evento.type == pygame.QUIT:			
                    se_ejecuta = False

                # Evento presionar teclas
                if evento.type == pygame.KEYDOWN:			
                
                    if evento.key == pygame.K_LEFT:			
                        jugador_x_cAmbio = -1			
                    if evento.key == pygame.K_RIGHT:		
                        jugador_x_cAmbio = 1
                    if evento.key == pygame.K_SPACE:		-> Barra espaciadora.
                        disparar_bala(jugador_x, bala_y)

                # Evento Soltar flechas
                if evento.type == pygame.KEYUP:
                    if  evento.key == pygame.K_LEFT or if evento.key == pygame.K_RIGHT:	
                        jugador_x_cAmbio = 0

            # Modificar ubicacion del jugador
            jugador_x += jugador_c_cambio

            # Mantener dentro de bordes del jugador
            if jugador_x <= 0:
                jugador_x = 0
            elif jugador_x >= 736:
                jugador_x = 736	

            # Modificar ubicacion del enemigo
            enemigo_x += enemigo_c_cambio

            # Mantener dentro de bordes del enemigo
            if enemigo_x <= 0:
                enemigo_x_cambio = 1
                enemigo_y += enemigo_y_cambio			
            elif enemigo_x >= 736:
                enemigo_x_cambio = -1
                enemigo_y += enemigo_y_cambio

            # Movimiento bala
            if bala_visible:
                disparar_bala(jugador_x, bala_y)
                bala_y -= bala_y_cambio			-> Perderá un 1 pixel por cada iteración e irá subiendo.
                    
                    
            jugador(jugador_x, jugador_y)					
            enemigo(enemigo_x, enemigo_y)					

            # Actualizar	
            pygame.display.update()

13. Detectar Colisiones

        Ahora toca calcular la distancia entre dos objetos en un plano bidimensional.
        Distancia entre los objetos es igual a la raíz cuadrada de la suma, del valor x del objeto2 menos el valor x del objeto1  al cuadrado, mas el valor 2 de nuestro objeto y menos el valor y de nuestro objeto 1, al cuadrado.
